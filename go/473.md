var res bool

func dfs(A []int,B []int,i int,target int) {
    if i >= len(A){
        for j:=1 ; j < 4 ; j++ {
            if B[j] != B[j-1] {
                return 
            }
        }
        
        res = true
        return 
    }
    
    if res {
        return
    }

    for j:=0; j < 4; j++ {
        if B[j] + A[i] <= target{
            B[j] += A[i];
            dfs(A,B,i+1,target);
            B[j]-=A[i]
        }
    }
}

    res = false
