'''
func generate(n int) [][]int {
    res := make([][]int, n)
    res[0]=make([]int, 1)
    res[0][0]=1;
    
    for i := 1; i < n; i++ {
        size := len(res[i-1])
        res[i]=make([]int, size+1)
        res[i][0]=1;
        for j:=1; j < size;j++{
            res[i][j]=res[i-1][j]+res[i-1][j-1];
        }
        res[i][size]=1;
    }
    return res;
}
