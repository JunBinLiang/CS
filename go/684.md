```func find(nums []int,x int) int{
    if nums[x] == x{
        return x
    }
    root := find(nums,nums[x])
    nums[x]=root
    return root
}


func findRedundantConnection(edges [][]int) []int {
    N := len(edges)
    nums := make([]int,N)
    
    for i:=0; i < N; i++{
        nums[i]=i
    }
    
    
    for _, edge := range edges{
        r1 := find(nums,edge[0]-1)
        r2 := find(nums,edge[1]-1)
        if r1 != r2 {
            nums[r1]=r2
        }else{
            return edge;
        }
    }
    return []int{0,0}
}```
