func most(A []int,bound int) int {
    res := 0
    mx := -1
    j := 0
    for i:=0; i < len(A); i++{
        if A[i] > mx {
            mx = A[i]
        }
        
        if mx > bound {
            mx = -1
            j = i+1
        }else{
            res += (i - j +1)
        }
    }
    return res
}

func numSubarrayBoundedMax(A []int, left int, right int) int {
    return most(A,right) - most(A,left-1);
}
