func check (s string, t string) bool {
    j :=0
    for i:=0; i < len(s); i++{
        if j >= len(t){
            return true;
        }
        if s[i] == t[j]{
            j++
        }
    }
    return j >= len(t)
}

func numMatchingSubseq(s string, words []string) int {
    res := 0
    for i:=0; i < len(words) ; i++{
        t := words[i]
        if check (s,t){
            res++
        }
    }
    return res
}
